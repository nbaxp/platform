# 目录结构

```txt
├── .vscode                         # vscode 项目配置
|   ├── extensions.json             # 插件列表
├── src                             # 源代码
|   ├── assets                      # 静态资源
|   ├── components                  # Vue 通用组件
|   ├── config                      # 配置文件
|   ├── layout                      # 布局组件
|   ├── lib                         # 三方组件
|   ├── locale                      # 国际化
|   ├── mock                        # 接口模拟
|   ├── models                      # 页面模型
|   ├── router                      # 路由配置
|   ├── store                       # 状态管理
|   ├── styles                      # 样式
|   ├── utils                       # 工具库
|   ├── views                       # 页面视图
|   ├── app.vue                     # app 组件
|   └──  main.js                    # js 入口
├── .editorconfig                   # 编辑器配置
├── .eslintignore                   # eslint 过滤配置
├── .eslintrc.json                  # javascript 代码检查
├── .gitattributes                  # git 文本配置
├── .gitignore                      # git 上传过滤
├── .markdownlint.json              # Markdown 代码检查
├── .markdownlintignore             # markdownlint 过滤配置
├── .prettierignore                 # prettier 过滤配置
├── .prettierrc.json                # 代码格式化配置
├── .stylelintignore                # stylelint 过滤配置
├── .stylelintrc.json               # CSS 代码检查
├── favicon.ico                     # 浏览器图标
├── index.html                      # web 入口
├── jsconfig.json                   # jsconfig 配置
├── package-lock.json               # 项目依赖
├── package.json                    # 项目配置
├── README.md                       # 项目说明
└── vite.config.js                  # vite 项目配置
```
